CATALOG MANAGEMENT
 
# 1/ Catalog : How to generate osmfile 
# Catalog Location : <WS>/<OS>/resources/graphic/CAAGSMFeaturesSplCatalog.CATfct
WS>mkrun -c cmd "CATFctEditorAssistant -client-id  SAMPLES -generate-osm CAAGSMFeaturesSplCatalog.CATfct CAAGSMFeaturesSplCatalog.osm"

# osm is generated under WS>CAAGSMFeaturesSplCatalog.osm
# Generate result is deliver as exemple in <WS>/Data.d/CAAGSMFeaturesSplCatalog.osm

# 2/ Catalog : How to upgrade catalog  
# modify osm file in order to extend Attribute SU , Add new SU in Catalog ,etc ... 
WS>mkrun -c cmd "CATFctEditorAssistant -client-id  SAMPLES -upgrade-catalog CAAGSMFeaturesSplCatalog.CATfct CAAGSMFeaturesSplCatalog.osm"

# catalog is generated under WS>CAAGSMFeaturesSplCatalog.CATfct
# Generated upgraded catalog has to be copied and  deliver in  <WS>/CAAGSMInterfaces.edu/CNext/resources/graphic/CAAGSMFeaturesSplCatalog.CATfct
